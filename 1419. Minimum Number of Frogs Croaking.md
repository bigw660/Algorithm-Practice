def minNumberOfFrogs(self, croakOfFrogs):
        # find cycles
        
        N = {"c":0, "r":1, "o":2, "a":3, "k":4}
        stack =[0]*5
        
        c = 1 # count
        for i, x in enumerate(croakOfFrogs[1:], 1):
            pre = N[croakOfFrogs[i-1]]
            if (N[x]-pre) == 1 or (N[x]-pre) == -4: # no gap
                continue
            else:
                if N[x] == 0:
                    stack[pre] += 1
                    if not stack[4]:
                        c += 1
                    else:
                        stack[4] -= 1
                else:
                    if stack[N[x]-1]:
                        stack[pre] += 1
                        stack[N[x]-1] -= 1
                    else:
                        return -1
                    
        return c if croakOfFrogs[-1]=="k" else -1
